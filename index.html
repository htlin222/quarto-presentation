// Configuration
const config = {
  apiKey: process.env.API_KEY,
};

// Async data fetching
async function fetchData(endpoint) {
  const response = await fetch(endpoint);
  const data = await response.json();
  return data;
}

// Main execution
fetchData("/api/users")
  .then((users) => console.log(users))
  .catch((err) => console.error(err));
